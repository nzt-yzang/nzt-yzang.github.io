进制转换

  • P1143 进制转换

常规方法
将一个10进制数转换为n进制数

 int main()
 {
     int n , k; 
     cin>>n>>k;
     string a;
     while(n){
         if(k > 10){
             a = (char) (n%k + 55)+a ;
         }
         else{
             a = (char) (n%k + '0')+a ;
         }
	     cout<<a<<endl;
	     n /= k;
     }

 }

 inline char cic(int x)//change into char 意在将一个n进制数的某位转成对应的m进制

##完整代码(将n进制数转换为m进制数)

 #include<cstdio>
 int n,x,m;
 inline int isint(char c)//可将'0'~'9'及'A'~'F'转化成如题所示数字并将非数字字符      返    回-1注意此时c并不会被修改
 {
     if (c >= 'A' && c <= 'F')
         return c - 55;//'A'=65,10=10;
     if (c > 47 && c < 58)
         return c - 48;
     return -1;
 }
 inline int read()
 {
     int x = 0;
     char c = getchar();
  while (isint(c) == -1) c = getchar();
  while (~isint(c)/*判断是不是数*/) x *= n, x += isint(c), c = getchar();
     return x;
 }//r
 inline char cic(int x) {
     if (x < 10)
         return x + 48;
     return x - 10 + 'A';
 }
 void otp(int k)
 {
     if (!k) return;
     otp(k / m);
     putchar(cic(k % m));
 }
 int main()
 {
     scanf("%d", &n);
     x = read();
     scanf("%d", &m);
     otp(x);
     putchar('\n');
 }
