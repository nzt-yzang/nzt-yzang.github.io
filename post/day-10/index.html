  • P1072 Hankson 的趣味题
    A的第一道数论蓝题(感觉有点水??)
    当然还是涨了一点知识的:
    gcd(a,b)=cgcd(a,b) = cgcd(a,b)=c 则gcd(a/c,b/c)=1gcd(a/c,b/c)=1gcd(a/c,b/c)=1,lcm(a,b)=clcm(a,b)=clcm(a,b)=c则gcd(c/b,c/a)=1gcd(c/b,c/a)=1gcd(c/b,c/a)=1
    即:两数最大公约数为c，即两数除以c后，得到的结果互质,同样，两数最小公倍数为c，即c除以两数之后，得到的结果互质
    关于这道题,可以发现x为b1的因数,所以枚举到sqrt(b1)就行,
#include <bits/stdc++.h>
using namespace std;
int n,a0,a1,b0,b1;
int gcd(int x,int y)
{
	if(y == 0) return x;
	return gcd(y,x%y);
}
int main()
{
	cin>>n;
	for(int i = 1 ; i <= n ; i++)
	{
		cin>>a0>>a1>>b0>>b1;
		int id = 0,sum = 0;
		for(int j = 1 ; j<=sqrt(b1) ; j++)
		{
			if(b1%j==0)
			{
				if(j%a1==0&&gcd(j/a1,a0/a1)==1&&gcd(b1/j,b1/b0)==1)
				{
					sum++; 	
				}
				if(b1!=j*j)
				if((b1/j)%a1==0&&gcd((b1/j)/a1,a0/a1)==1&&gcd(b1/(b1/j),b1/b0)==1)
				{
					sum++; 	
				}
			}
		}
		cout<<sum<<endl;
	}
	return 0;
}
  • P1827 [USACO3.4]美国血统 American Heritage
    已知二叉树中序和前序遍历求后序的题,
    首先应该清楚中序遍历是(左,根,右),前序遍历是(根,左,右),后序遍历是(左,右,根);
    然后递归就行
 #include <bits/stdc++.h>
using namespace std;
string a,b;
void dfs(int x,int y,int p,int q)
{
	if(x>y||p>q) return;
	else{
		int i = a.find(b[x]);//前序遍历第一个是根,在后序遍历中找到根的位置, 
	dfs(x+1,x+i-p,p,i-1);
	dfs(x+i-p+1,y,i+1,q);
	cout<<b[x];
	} 
}
int main()
{
	cin>>a>>b;
	int l = a.length() - 1;
	dfs(0,l,0,l);
	return 0;
}
