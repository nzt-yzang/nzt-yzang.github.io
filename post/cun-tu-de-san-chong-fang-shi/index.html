       for(int i=0; i<=n; i++)
           v[i].clear();//将vecort数组清空
       for(int i=1; i<=m; i++) //用vector存储邻接表
       {
           node nd;
           scanf("%d%d%d",&a,&b,&c);
           nd.d=b,nd.w=c;//将入度的点和权值赋值给结构体
           v[a].push_back(nd);//将每一个从a出发能直接到达的点都压到下标为a的vector数组中，以后遍历从a能到达的点就可以直接遍历v[a]
           nd.d=a,nd.w=c;//无向图的双向存边
           v[b].push_back(nd);
       }

3,链式前向星

int head[maxn<<1];
struct Edge{
    int v,w,next;
}edge[maxn];
void _addedge(int u , int v , int w)
{
    edge[++cnt].next = head[u];
    edge[cnt].v = v;
    edge[cnt].w = w;
    head[u] = cnt;
}
